## 🚨 PROBLEM 1: False Attribution (CRITICAL)

### Current Flaw:
```python
# BAD: Assumes pattern helped just because success happened
if success:
    for sig in injected_pattern_signatures:
        increment_helped_count(sig)  # ❌ Correlation, not causation
```

**Problem:** LLM might fix correctly WITHOUT using pattern, or pattern was irrelevant.

### Fix: Track Pattern Usage Signal

```python
# GOOD: Verify pattern was actually used
def _check_pattern_usage(pattern: FailurePattern, patch_diff: str) -> bool:
    """
    Check if pattern's guidance was actually followed in the patch.

    Returns True if patch shows evidence of following pattern's correct_direction.
    """
    if not pattern.correct_direction:
        return False

    patch_lower = patch_diff.lower()
    direction_lower = pattern.correct_direction.lower()

    # Extract key action words from correct_direction
    action_keywords = []
    if "import" in direction_lower:
        action_keywords.append("import")
    if "check" in direction_lower and "type" in direction_lower:
        action_keywords.extend(["isinstance", "type(", "Array.isArray"])
    if "loading" in direction_lower or "state" in direction_lower:
        action_keywords.extend(["loading", "isLoading", "useState"])
    if "parent" in direction_lower or "component" in direction_lower:
        # Check if modified correct file
        if pattern.context and "error_file" in pattern.context:
            error_file = pattern.context["error_file"].lower()
            return error_file in patch_lower

    # Check if ANY action keyword appears in patch
    return any(kw in patch_lower for kw in action_keywords)

# Updated tracking:
if success and injected_pattern_signatures:
    for sig in injected_pattern_signatures:
        pattern = memory._store._by_signature[sig].pattern
        if _check_pattern_usage(pattern, patch_diff):
            memory._store._increment_helped_count(sig)  # ✅ Causal
        else:
            logger.debug(f"Pattern {sig[:8]} injected but not used")
```

## 🚨 PROBLEM 2: Cold Start Noise (CRITICAL)

### Current Flaw:
```python
# BAD: 40% weight on noisy early data
score = 0.4 * effectiveness  # effectiveness = 1/1 = 100% after 1 success!
```

**Problem:** Early lucky patterns get overweighted, new patterns ignored.

### Fix: Bayesian Smoothing + Minimum Threshold

```python
def _compute_effective_score(entry: FailureEntry) -> float:
    """
    Compute effectiveness with Bayesian smoothing and minimum threshold.

    - If times_injected < 5: Use low weight (0.1)
    - Otherwise: Use Bayesian smoothed effectiveness
    """
    if entry.times_injected < 5:
        # Cold start: low weight, use frequency as proxy
        return 0.1 * min(1.0, entry.frequency / 10)

    # Bayesian smoothing: (successes + 1) / (attempts + 2)
    # This prevents 1/1 = 100% and 0/1 = 0% extremes
    smoothed_effectiveness = (entry.times_helped + 1) / (entry.times_injected + 2)

    return smoothed_effectiveness

# Updated retrieval scoring:
def _compute_relevance(..., entry: FailureEntry):
    # Use smoothed effectiveness instead of raw
    effective_score = _compute_effective_score(entry)
    score += 0.4 * effective_score  # ✅ Anti-noise
```

## 🚨 PROBLEM 3: Soft Constraints (HIGH)

### Current Flaw:

**Problem:** LLM "understands but ignores" - no forced acknowledgment.

### Fix: Forced Reasoning Hook

```python
def format_for_prompt_v3(relevant_failures):
    """v3: Add forced reasoning hook"""

    lines = ["[FAILURE MEMORY - CRITICAL CONSTRAINTS]"]
    lines.append("You are making a repair to code that previously failed.\n")

    # Show patterns
    for i, (pattern, lesson, score) in enumerate(relevant_failures, 1):
        lines.append(f"{i}. {pattern.cause.upper()}")
        if pattern.attempted_fix:
            lines.append(f"   Previous attempt: {pattern.attempted_fix}")
        if pattern.correct_direction:
            lines.append(f"   Why it failed: {pattern.correct_direction}")
        lines.append(f"   Confidence: {lesson.confidence:.0%}\n")

    # FORCED REASONING HOOK
    lines.append("BEFORE WRITING THE FIX, YOU MUST:")
    lines.append("1. Identify which failure pattern(s) apply to this error")
    lines.append("2. Explain why the previous fix approach failed")
    lines.append("3. State what you will do differently this time")
    lines.append("4. THEN write the patch\n")

    lines.append("STRICT RULES:")
    for i, (pattern, lesson, score) in enumerate(relevant_failures, 1):
        lines.append(f"  {i}. {lesson.avoid}")
        lines.append(f"     → MUST: {lesson.prefer}")

    lines.append("\nIf you skip the reasoning or violate these rules, the patch will fail again.")

    return "\n".join(lines)
```

```python
def _rule_wrong_fix_strategy(self, surface, error, patch, files):
    has_function_error = "is not a function" in error
    has_null_check = any(kw in patch for kw in ["if (", "null", "undefined"])

    if has_function_error and has_null_check:
        func_match = re.search(r"(\w+) is not a function", surface.error_text)
        func_name = func_match.group(1) if func_match else "function"

        # SUB-TYPE based on context
        if any(f.endswith('.jsx') or f.endswith('.tsx') for f in files):
```

```python
    # Track if this is first attempt
    is_first_attempt = (repair_count == 0)

        for sig in injected_pattern_signatures:
            memory._store._increment_first_pass_attempt(sig)

            for sig in injected_pattern_signatures:
                if _check_pattern_usage(pattern, patch_diff):
                    memory._store._increment_first_pass_success(sig)

# Compute metrics:
def get_pattern_metrics(entry: FailureEntry) -> dict:
    """Compute comprehensive pattern metrics"""
    metrics = {
        "effectiveness": entry.effectiveness,
        "first_pass_rate": 0.0,
        "retry_reduction": 0.0
    }

    if entry.total_first_pass_attempts > 0:
        metrics["first_pass_rate"] = (
            entry.first_pass_successes / entry.total_first_pass_attempts
        )

    # Compare to baseline (assume 30% first-pass without pattern)
    baseline_first_pass = 0.3
    if metrics["first_pass_rate"] > baseline_first_pass:
        metrics["retry_reduction"] = (
            (metrics["first_pass_rate"] - baseline_first_pass) / baseline_first_pass
        )

    return metrics
```
